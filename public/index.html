AI Cookie Consent Widget

A lightweight, customizable cookie consent widget with AI features.

Quick Start

Add these scripts to your website:

<script src="config.js"></script>
<script src="https://ai-cookie-consent.vercel.app/cookie-consent.min.js"></script>

Basic Usage

<script>
    // Initialize with config settings
    window.cookieConsent.init({
        apiKey: window.CONFIG.OPENAI_API_KEY,
        ...window.CONFIG.WIDGET_SETTINGS,
        onAccept: function() {
            console.log('Cookies accepted!');
        },
        onDecline: function() {
            console.log('Cookies declined!');
        }
    });
</script>

Advanced Configuration

window.cookieConsent.init({
    position: 'bottom-right', // or 'bottom-left', 'top-right', 'top-left'
    theme: 'dark', // or 'light'
    message: 'Custom message here',
    acceptButtonText: 'Accept',
    declineButtonText: 'Decline',
    onAccept: function() {
        console.log('Cookies accepted!');
    },
    onDecline: function() {
        console.log('Cookies declined!');
    }
});
